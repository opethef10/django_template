    {% translate "Environment Variables" %}

    {% translate "All sensitive configuration is managed via environment variables using" %} python-decouple.

    {% translate "Required Variables" %}

    # Project Identity
PROJECT_SLUG=myproject
PROJECT_NAME=My Project
PROJECT_DESCRIPTION=A brief description of my project
PROJECT_DOMAIN=myproject.com
PROJECT_ADMIN_NAME=Admin
PROJECT_ADMIN_EMAIL=admin@myproject.com
DJANGO_SECRET_KEY=your-very-long-random-secret-key

    {% translate "Optional Variables" %}

    # Email (for contact form and notifications)
DJANGO_EMAIL_ENABLED=True
DJANGO_SERVER_EMAIL=noreply@myproject.com
DJANGO_EMAIL_HOST_USER=smtp@gmail.com
DJANGO_EMAIL_HOST_PASSWORD=your-app-password

# Security
DJANGO_ALLOWED_HOSTS=myproject.com,www.myproject.com

# Account Settings
ACCOUNT_ALLOW_SIGNUPS=True  # Set to False to disable public registration

# reCAPTCHA (for signup spam protection)
RECAPTCHA_ENABLED=True
RECAPTCHA_PUBLIC_KEY=your-recaptcha-site-key
RECAPTCHA_PRIVATE_KEY=your-recaptcha-secret-key

    {% translate "Development Settings" %}

    {% translate "When running locally, the following are automatically enabled:" %}

    • DEBUG = True
    • ALLOWED_HOSTS = ["*"]
    • Debug Toolbar - Available at /__debug__/
    • Console Email Backend - Emails printed to terminal instead of sent
    • Dummy Cache - No actual caching

    {% translate "Production Settings" %}

    • DEBUG = False
    • ALLOWED_HOSTS from environment variable
    • File Logging - Logs to /var/www/proj.log
    • Static/Media - Served from /var/www/static and /var/www/media
    • Real SMTP - Only if DJANGO_EMAIL_ENABLED=True
