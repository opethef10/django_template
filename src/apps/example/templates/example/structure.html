    {% translate "App Structure" %}

    {% translate "Each app in" %} src/apps/ {% translate "follows this pattern:" %}

    src/apps/<app_name>/
├── __init__.py
├── apps.py                   # AppConfig class
├── models.py                 # Database models
├── views.py                  # Class-based views
├── urls.py                   # App URLs (with app_name)
├── admin.py                  # Django admin registration
├── forms.py                  # Form classes
├── signals.py                # Django signals (optional)
├── utils.py                  # Helper functions (optional)
├── templates/<app_name>/    # App-specific templates
│   └── *.html
├── static/<app_name>/        # App-specific static files
├── migrations/               # Database migrations
│   ├── __init__.py
│   └── 0*.py
└── tests/                    # Test files
    ├── __init__.py
    └── test_*.py

    {% translate "Adding a New App" %}

    {% translate "1. Create the app directory" %}

    mkdir -p src/apps/myapp

    {% translate "2. Create the required files" %}

    # __init__.py (empty)
# apps.py
from django.apps import AppConfig

class MyappConfig(AppConfig):
    default_auto_field = "django.db.models.BigAutoField"
    name = "src.apps.myapp"

# urls.py
from django.urls import path

app_name = "myapp"
urlpatterns = [
    path("", TemplateView.as_view(template_name="myapp/index.html"), name="index"),
]

    {% translate "3. Register in settings" %}

    {% translate "Add to" %} INSTALLED_APPS {% translate "in" %} settings/_base.py:

    INSTALLED_APPS = [
    # ...
    "src.apps.myapp",
]

    {% translate "4. Include URLs in root" %}

    {% translate "Add to" %} src/urls.py:

    path("myapp/", include("src.apps.myapp.urls"))

    {% translate "5. Create templates" %}

    mkdir -p src/apps/myapp/templates/myapp
