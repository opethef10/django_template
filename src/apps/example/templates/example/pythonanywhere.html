    • {% translate "Domain name (optional but recommended)" %}

    {% translate "Step 1: Set Up PythonAnywhere Console" %}

    1.1 {% translate "Open Bash Console" %}

    {% translate "Log in to PythonAnywhere and open a" %} Bash console.

    1.2 {% translate "Clone Your Repository" %}

    git clone https://github.com/yourusername/yourproject.git
cd yourproject

    1.3 {% translate "Create Virtual Environment" %}

    python -m venv venv
source venv/bin/activate
pip install -r requirements/production.txt

    {% translate "Step 2: Configure Environment Variables" %}

    2.1 {% translate "Create .env File" %}

    nano .env

    {% translate "Add the following (adjust values for your project):" %}

    PROJECT_SLUG=yourproject
PROJECT_NAME=Your Project
PROJECT_DESCRIPTION=Your project description
PROJECT_DOMAIN=yourusername.pythonanywhere.com
PROJECT_ADMIN_NAME=Admin
PROJECT_ADMIN_EMAIL=your@email.com
DJANGO_SECRET_KEY=generate-a-secure-random-key
DJANGO_ALLOWED_HOSTS=yourusername.pythonanywhere.com
DJANGO_EMAIL_ENABLED=False
ACCOUNT_ALLOW_SIGNUPS=True

    2.2 {% translate "Generate a Secret Key" %}

    python -c "from django.core.management.utils import get_random_secret_key; print(get_random_secret_key())"

    {% translate "Step 3: Run Database Migrations" %}

    python manage.py migrate

    {% translate "Create Admin User" %}

    python manage.py createsuperuser

    {% translate "Step 4: Collect Static Files" %}

    mkdir -p /var/www/static /var/www/media
python manage.py collectstatic

    {% translate "Step 5: Configure Web App" %}

    5.1 {% translate "Create WSGI File" %}

    {% translate "Use the provided script:" %}

    ./scripts/create_wsgi.sh

    {% translate "This creates" %} /var/www/<username>_pythonanywhere_com_wsgi.py

    5.2 {% translate "Configure in PythonAnywhere Dashboard" %}

    1. {% translate "Go to" %} Web {% translate "tab" %}
    2. {% translate "Click" %} Add a new web app
    3. {% translate "Select your domain" %}
    4. {% translate "Choose" %} Manual configuration
    5. {% translate "Select Python" %} 3.10+
    6. {% translate "Set Virtualenv path to" %} /home/<username>/<project>/venv

    5.3 {% translate "Set WSGI File" %}

    {% translate "In the Web tab, set the WSGI file to:" %}

    /var/www/<username>_pythonanywhere_com_wsgi.py

    5.4 {% translate "Configure Static Files" %}

    {% translate "Add static file mapping:" %}

    {% translate "URL" %} {% translate "Directory" %}
    /static/ /var/www/static
    /media/ /var/www/media

    {% translate "Step 6: Reload and Test" %}

    {% translate "Click the" %} Reload {% translate "button in the Web tab, then visit your site." %}

    {% translate "Updating Your Deployment" %}

    {% translate "For future updates, run in your Bash console:" %}

    cd ~/yourproject
git pull
source venv/bin/activate
pip install -r requirements/production.txt
python manage.py migrate
python manage.py collectstatic --noinput

    {% translate "Then reload from the Web tab." %}

    {% translate "Using Deployment Scripts" %}

    {% translate "For automated updates:" %}

    # Full update (backup + pull + install + migrate + collectstatic + reload)
./scripts/update.sh

# Quick reload only
./scripts/reload.sh

    {% translate "Troubleshooting" %}

    • {% translate "Check PythonAnywhere error log in Web tab" %}
    • {% translate "Verify .env file exists and has correct values" %}
    • {% translate "Check that virtualenv path is correct" %}
    • {% translate "Ensure STATIC_ROOT is set correctly" %}

    • {% translate "Run" %} python manage.py collectstatic
    • {% translate "Check static file mapping in Web tab" %}
    • {% translate "Verify directory permissions (chmod 755)" %}

    • {% translate "Run" %} python manage.py migrate
    • {% translate "Check database file permissions" %}
    • {% translate "Verify DATABASE_NAME in settings" %}
